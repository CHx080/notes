# 最大正方形

https://leetcode.cn/problems/maximal-square/description/

```c++
int maximalSquare(vector<vector<char>>& matrix) {
    int m = matrix.size(), n = matrix[0].size();
    vector<vector<int>> dp(m, vector<int>(n, 0));  /* dp(i,j)以(i,j)为右下角的最大正方形边长 */
    int maxSide = 0;  
    for (int i = 0; i < m; ++i) {
        dp[i][0] = matrix[i][0] - '0';
        maxSide = max(maxSide, dp[i][0]);
    }
    for (int j = 0; j < n; ++j) {
        dp[0][j] = matrix[0][j] - '0';
        maxSide = max(maxSide, dp[0][j]);
    }
    for (int i = 1; i < m; ++i) 
        for (int j = 1; j < n; ++j) 
            if (matrix[i][j] == '1') {
                dp[i][j] = min({dp[i-1][j], dp[i][j-1], dp[i-1][j-1]}) + 1;
                maxSide = max(maxSide, dp[i][j]);
            }
    return maxSide * maxSide;  
}
```

# 统计全1方阵

https://leetcode.cn/problems/count-square-submatrices-with-all-ones/description/

```c++
int countSquares(vector<vector<int>>& matrix) {
    int m = matrix.size(), n = matrix[0].size();
    vector<vector<int>> dp(m, vector<int>(n, 0));  
    /* dp(i,j)以(i,j)为右下角的最大正方形边长 */
    int cnt = 0;  
    for (int i = 0; i < m; ++i) {
        dp[i][0] = matrix[i][0];
        if(matrix[i][0]) ++cnt;
    }
    for (int j = 1; j < n; ++j) {
        dp[0][j] = matrix[0][j];
        if(matrix[0][j]) ++cnt;
    }
    for (int i = 1; i < m; ++i)
        for (int j = 1; j < n; ++j)
            if (matrix[i][j] == 1) {
                dp[i][j] = min({dp[i-1][j], dp[i][j-1], dp[i-1][j-1]}) + 1;
                cnt+=dp[i][j];
            }
    return cnt;
}
```

# 最大边界1方阵

https://leetcode.cn/problems/largest-1-bordered-square/description/

```c++
int largest1BorderedSquare(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    vector<vector<int>> left(m, vector<int>(n, 0));  // 记录每个位置左侧连续1的个数
    vector<vector<int>> top(m, vector<int>(n, 0));   // 记录每个位置上方连续1的个数
    int maxLen = 0;  
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (grid[i][j] == 1) {
                left[i][j] = (j == 0) ? 1 : left[i][j - 1] + 1;  
                top[i][j] = (i == 0) ? 1 : top[i - 1][j] + 1;   
            }
        }
    }

    for (int i = m - 1; i >= 0; --i) {
        for (int j = n - 1; j >= 0; --j) {
            int len = min(left[i][j], top[i][j]);  // 当前可能的最大边长
            while (len > maxLen) {
                // 检查右下角是否满足条件
                if (left[i - len + 1][j] >= len && top[i][j - len + 1] >= len) {
                    maxLen = len;  // 更新最大边长
                    break;
                }
                len--;  // 缩小边长继续检查
            }
        }
    }
    return maxLen * maxLen;  
}
```

# 最大加号标志

https://leetcode.cn/problems/largest-plus-sign/description/

```c++
int orderOfLargestPlusSign(int n, vector<vector<int>>& mines) {
        vector<vector<int>> grid(n, vector<int>(n, 1));
        for (auto& mine : mines) grid[mine[0]][mine[1]] = 0;
        vector<vector<int>> left(n, vector<int>(n, 0));
        auto right=left,up=left,down=left;  //
        for (int i = 0; i < n; ++i) 
            for (int j = 0; j < n; ++j) 
                if (grid[i][j] == 1) {
                    left[i][j] = (j == 0) ? 1 : left[i][j - 1] + 1;
                    up[i][j] = (i == 0) ? 1 : up[i - 1][j] + 1;
                }
        for (int i = n - 1; i >= 0; --i) 
            for (int j = n - 1; j >= 0; --j) 
                if (grid[i][j] == 1) {
                    right[i][j] = (j == n - 1) ? 1 : right[i][j + 1] + 1;
                    down[i][j] = (i == n - 1) ? 1 : down[i + 1][j] + 1;
                }
        int maxOrder = 0;
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j) 
                if (grid[i][j] == 1) {
                    int order =min({left[i][j], right[i][j], up[i][j], down[i][j]});
                    maxOrder = max(maxOrder, order);
                }
        return maxOrder;
    }
```
